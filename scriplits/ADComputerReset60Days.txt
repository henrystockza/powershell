Import-Module *Active*
Get-ADComputer -Filter {UserPrinciplieName -like "**"}  -prop * | Select isDeleted,enabled,CN,displayName,description, @{n="lastLogonDate";e={[datetime]::FromFileTime($_.lastLogonTimestamp)}},@{n="pwdLastSet";e={[datetime]::FromFileTime($_.pwdLastSet)}},sid |  where{$_.pwdLastSet -ge (get-date).adddays(-60)} |   export-csv ADComputers.csv
